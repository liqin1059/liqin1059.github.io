var lengthOfLongestSubstring = function(s) {
    let arr = [], max = 0;
    for (let i = 0; i < s.length; i++) {
        // 查找是否已存在字符
        let index = arr.indexOf(s[i]);
        // 如果存在，则删除查找到的相同字符的位置以及之前的字符
        if(index !== -1) {
          arr.splice(0, index + 1);
        }
        // 当前字符添加到数组中
        arr.push(s.charAt(i));
        max = Math.max(arr.length, max);
    }
    return max;
};
