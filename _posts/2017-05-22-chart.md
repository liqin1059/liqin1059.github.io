//页面加载完执行的多个函数
function addLoadEvent(func) {
    var ages = [].slice.call(arguments,1);
    var oldonload = window.onload;
    if (typeof window.onload != 'function') {
        window.onload = function() {
            func.apply(this,ages);
        }
    } else {
        window.onload = function() {
            oldonload.apply(this);
            func.apply(this,ages);
        }
    }
}
